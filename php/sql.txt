-- Tabla usuario
CREATE TABLE usuario (
    id_usuario INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100),
    email VARCHAR(100),
    password VARCHAR(100),
    pais VARCHAR(100),
    ciudad VARCHAR(100),
    cedula VARCHAR(100),
    telefono VARCHAR(100),
    direccion TEXT,
    fotoperfil LONGBLOB,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- Tabla empresa
CREATE TABLE empresa (
    id_empresa INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100),
    correo VARCHAR(100),
    pais VARCHAR(100),
    telefono VARCHAR(100),
    direccion TEXT,
    password VARCHAR(100),
    fotoperfil LONGBLOB,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- Tabla hotel
CREATE TABLE hotel (
    id_hotel INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100),
    pais VARCHAR(100),
    ciudad VARCHAR(100),
    telefono VARCHAR(100),
    direccion TEXT,
    id_empresa INT,
    foto LONGBLOB,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (id_empresa) REFERENCES empresa(id_empresa)
);

-- Tabla restaurante
CREATE TABLE restaurante (
    id_restaurante INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100),
    pais VARCHAR(100),
    ciudad VARCHAR(100),
    telefono VARCHAR(100),
    direccion TEXT,
    id_empresa INT,
    foto LONGBLOB,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (id_empresa) REFERENCES empresa(id_empresa)
);

-- Tabla favoritos de usuario para restaurantes
CREATE TABLE usuario_favorito_restaurante (
    id_usuario INT,
    id_restaurante INT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_usuario) REFERENCES usuario(id_usuario),
    FOREIGN KEY (id_restaurante) REFERENCES restaurante(id_restaurante),
    PRIMARY KEY (id_usuario, id_restaurante)
);

-- Tabla favoritos de usuario para hoteles
CREATE TABLE usuario_favorito_hotel (
    id_usuario INT,
    id_hotel INT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_usuario) REFERENCES usuario(id_usuario),
    FOREIGN KEY (id_hotel) REFERENCES hotel(id_hotel),
    PRIMARY KEY (id_usuario, id_hotel)
);

CREATE TABLE usuario_historial_busqueda (
    id_historial INT AUTO_INCREMENT PRIMARY KEY,
    id_usuario INT,
